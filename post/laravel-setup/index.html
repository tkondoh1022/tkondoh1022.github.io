Setup WorkDir

1$ mkdir -p  work-github/laravel
2$ cd work-github/laravel
3$ echo "# laravel" >> README.md
4$ git init
5$ git add README.md
6$ git commit -m "first commit"
7$ git remote add origin https://github.com/tkondoh1022/laravel.git
8$ git push -u origin master
