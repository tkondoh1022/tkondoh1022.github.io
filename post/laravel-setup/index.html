Setup alias

1$ grep artisan ~/.bashrc
2alias artisan='docker-compose exec -u tkondoh app php artisan'
3alias composer='docker-compose exec app composer'
4$ . ~/.bashrc

Create Sample Project

 1$ composer create-project --prefer-dist laravel/laravel=6.* .
 2$ composer require predis/predis
 3
 4$ tree ./src-lsapp
 5./src-lsapp
 6    ├── .editorconfig
 7    ├── .env       ... 環境設定情報
 8    ├── .env.example
 9    ├── app/       ... アプリケーションのロジック
10    ├── artisan    ... laravel のコマンドラインツール
11    ├── bootstrap/ ... laravel フレームワークの起動コード
12    ├── composer.json
13    ├── composer.lock
14    ├── config/    ... 設定ファイル
15    ├── database/  ... Migration ファイルなど DB 関連
16    ├── package.json
17    ├── phpunit.xml
18    ├── public/    ... Web サーバのドキュメントルート
19    ├── resources/ ... ビューや言語変換用ファイルなど
20    ├── routes/    ... ルーティング用ファイル
21    ├── server.php
22    ├── storage/   ... フレームワークが使用するファイル
23    ├── tests/     ... テストコード
24    └── vendor/    ... Composer でインストールしたライブラリ
